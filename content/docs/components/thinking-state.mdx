## Integration Example

Here's how to integrate the ThinkingState component in an AI chat interface:

```tsx
import { ThinkingState } from "@/components/haber-ui/thinking-state";
import { useState } from "react";

export function ChatInterface() {
  const [isProcessing, setIsProcessing] = useState(false);

  const handleSubmit = async (message) => {
    setIsProcessing(true);
    try {
      // Process message with AI
      await processWithAI(message);
    } finally {
      setIsProcessing(false);
    }
  };

  return (
    <div className="space-y-4">
      <div className="h-[400px] overflow-y-auto">{/* Chat messages */}</div>

      {isProcessing && (
        <ThinkingState
          messages={[
            "Thinking...",
            "Processing your request...",
            "Generating response...",
          ]}
        />
      )}

      <form onSubmit={handleSubmit}>{/* Input field and submit button */}</form>
    </div>
  );
}
```
